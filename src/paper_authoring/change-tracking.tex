% Change-tracking macros for paper_authoring workflow.
% Requires: xcolor, changebar, marginnote packages.
% Note: does NOT require soul/\sout — deleted text is greyed out, not struck through.
% This allows \deleted and \replaced to work across paragraph boundaries.

% Inline change markup
\definecolor{addedcolor}{RGB}{0,160,0}
\definecolor{deletedcolor}{RGB}{180,180,180}
\newcommand{\added}[1]{\textcolor{addedcolor}{#1}}
\newcommand{\deleted}[1]{\textcolor{deletedcolor}{#1}}
\newcommand{\replaced}[2]{\textcolor{deletedcolor}{#2}\textcolor{addedcolor}{#1}}

% Change bars for human review — red: proposed edit; blue: task selection.
% Workaround: changebar's \end@dblfloat redefinition breaks labels in table* floats.
% Wrap table* environments with \cboff/\cbon to temporarily restore originals.
\makeatletter
\newcommand{\cboff}{\let\end@dblfloat\ltx@end@dblfloat}
\newcommand{\cbon}{\let\end@dblfloat\cb@end@dblfloat}
\makeatother
\newcommand{\reviewstart}{\cbcolor{red}\cbstart\marginnote{\parbox{1.2cm}{\scriptsize\itshape\textcolor{red}{Approval\\needed}}}}
\newcommand{\reviewend}{\cbend}
\newcommand{\editstart}{\cbcolor{blue}\cbstart\marginnote{\parbox{1.2cm}{\scriptsize\itshape\textcolor{blue}{In\\progress}}}}
\newcommand{\editend}{\cbend}
